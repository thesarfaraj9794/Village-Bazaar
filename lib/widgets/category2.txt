import 'package:cached_network_image/cached_network_image.dart';
import 'package:cloud_firestore/cloud_firestore.dart';
import 'package:flutter/cupertino.dart';
import 'package:flutter/material.dart';
import 'package:get/get.dart';
import 'package:village_bazzar/models/category-model.dart';
import 'package:village_bazzar/screens/user-panel/single-category-product-screen.dart';

class CategoriesWidget extends StatelessWidget {
  const CategoriesWidget({super.key});

  @override
  Widget build(BuildContext context) {
    return FutureBuilder(
      future: FirebaseFirestore.instance.collection('categories').get(),
      builder: (BuildContext context, AsyncSnapshot<QuerySnapshot> snapshot) {
        if (snapshot.hasError) {
          return Center(child: Text('Error loading categories'));
        }
        if (snapshot.connectionState == ConnectionState.waiting) {
          // Adjust height for the expected category display
          return SizedBox(
            height: Get.width / 2.5, // Approx height for 2 rows of items
            child: const Center(child: CupertinoActivityIndicator()),
          );
        }
        if (snapshot.data!.docs.isEmpty) {
          return const Center(child: Text('No categories available'));
        }

        List<CategoriesModel> categories = snapshot.data!.docs.map((doc) {
          return CategoriesModel(
            categoryId: doc['categoryId'],
            categoryImg: doc['categoryImg'],
            categoryName: doc['categoryName'],
            createdAt: doc['createdAt'],
            updatedAt: doc['updatedAt'],
          );
        }).toList();

        // **संशोधित डिज़ाइन यहाँ से शुरू होता है**
        return Container(
          // Set a height suitable for 2 rows of items + padding/spacing
          height: Get.width / 2.5 + 20, // Example height
          // You can remove the background color or set it to transparent/white
          // color: Color.fromARGB(255, 190, 207, 243),
          padding: const EdgeInsets.symmetric(horizontal: 10.0, vertical: 5.0),
          child: GridView.builder(
            scrollDirection: Axis.horizontal, // क्षैतिज स्क्रॉलिंग
            gridDelegate: SliverGridDelegateWithFixedCrossAxisCount(
              crossAxisCount: 2, // 2 rows
              mainAxisSpacing: 15, // Horizontal spacing between items
              crossAxisSpacing: 10, // Vertical spacing between items/rows
              childAspectRatio: 1.15, // Adjust aspect ratio to fit content (icon + text) better
            ),
            itemCount: categories.length,
            itemBuilder: (context, index) {
              CategoriesModel category = categories[index];
              return GestureDetector(
                onTap: () => Get.to(
                  () => AllSingleCategoryProductsScreen(
                    categoryId: category.categoryId,
                  ),
                ),
                child: Column(
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: [
                    // Icon/Image container
                    Container(
                      width: Get.width / 7, // Set width for the circular background
                      height: Get.width / 7, // Set height for the circular background
                      decoration: BoxDecoration(
                        color: const Color.fromARGB(255, 237, 246, 236), // Light green/background color
                        borderRadius: BorderRadius.circular(15), // Rounded corners for a squarish look like in the image
                        boxShadow: [
                          BoxShadow(
                            color: Colors.grey.withOpacity(0.15),
                            spreadRadius: 1,
                            blurRadius: 3,
                            offset: const Offset(0, 2),
                          ),
                        ],
                      ),
                      child: ClipRRect(
                        borderRadius: BorderRadius.circular(15),
                        child: Padding(
                          padding: const EdgeInsets.all(5.0),
                          child: CachedNetworkImage(
                            imageUrl: category.categoryImg,
                            fit: BoxFit.contain, // Use BoxFit.contain to prevent image cropping
                            placeholder: (context, url) => const Center(child: CupertinoActivityIndicator()),
                            errorWidget: (context, url, error) => const Icon(Icons.error),
                          ),
                        ),
                      ),
                    ),
                    const SizedBox(height: 6), // Spacing between icon and text
                    // Category Name
                    Text(
                      category.categoryName,
                      textAlign: TextAlign.center,
                      maxLines: 1,
                      overflow: TextOverflow.ellipsis,
                      style: const TextStyle(
                        fontWeight: FontWeight.w600, // Slightly bold text
                        fontSize: 12,
                        color: Colors.black87,
                      ),
                    ),
                  ],
                ),
              );
            },
          ),
        );
        // **संशोधित डिज़ाइन यहाँ समाप्त होता है**
      },
    );
  }
}
